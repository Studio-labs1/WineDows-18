      'images/background4.png'

    // Функция для установки случайного фона
    function setRandomBackground() {
      const randomIndex = Math.floor(Math.random() * backgrounds.length);
      const randomBackground = backgrounds[randomIndex];
      document.body.style.backgroundImage = `url('${randomBackground}')`;

    // Обновление времени и даты
    function updateClock() {
      const now = new Date();
      
      // Форматируем время: чч:мм
      const hours = String(now.getHours()).padStart(2, '0');
      const minutes = String(now.getMinutes()).padStart(2, '0');
      const timeStr = `${hours}:${minutes}`;
      
      // Форматируем дату: ДД.ММ.ГГГГ
      const day = String(now.getDate()).padStart(2, '0');
      const month = String(now.getMonth() + 1).padStart(2, '0'); // Месяц от 0 до 11
      const year = now.getFullYear();
      const dateStr = `${day}.${month}.${year}`;
      
      // Выводим в панель
      document.getElementById('clock').textContent = `${timeStr}  ${dateStr}`;
    }

    // Открытие сайтов по двойному клику + установка фона и часов при загрузке
